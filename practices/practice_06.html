import numpy as np
import matplotlib.pyplot as plt

def bbplank(l, T: float):
    k = 1.38e-23 
    c = 3e8
    h = 6.62e-34
    result = 2*h*c**2/(l**5)*1/(np.exp(h*c/(l*k*T))-1)
    return result

scale = 1e-9
x = scale * np.linspace(250, 12000, 100)

objects = {'Sun': 5250 + 273, 'Metall 1200': 1200, 'Human': 37 + 273}

objects['2700 LED'] = 2700

collect = []
for item, temp in objects.items():
    i = bbplank(x, temp)
    plt.plot(x, i, label=item)
    collect.append(i)
    #plt.plot(x, i, label=item)

plt.yscale('log')
plt.yscale('log')
plt.ylim(10e2, 10e14)
plt.legend()
plt.show()
