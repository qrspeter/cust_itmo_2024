def train(training_set_inputs,training_set_outputs):
    #print(len(training_set_inputs))
    #print(training_set_inputs)
    synaptic_weights = np.empty((100,1))
    #synaptic_weights = np.array([[0.], [0.], [0.]]) 
    l_rate = 0.2
    trainings = 50
    
    for iteration in range(trainings): 
        output = 1/(1+np.exp(-np.dot(training_set_inputs, synaptic_weights)))
        error = training_set_outputs - output.T
        #print(error)
        #print(training_set_outputs)
        #print(output)
        synaptic_weights += np.dot(training_set_inputs.T,l_rate*error.T)
    return synaptic_weights
