import numpy as np
import matplotlib.pyplot as plt

l = np.linspace(3e-7,7e-6,100)
T = 2800
def B(l,T):     
    c = 3e8
    h = 6.63e-34
    k = 1.38e-23
    i = ((2*h*c**2)/(l**5))/(np.exp(((h*c)/(l*k*T)))-1)
    
    return i
   
plt.plot(l, B(l,T), 'r')
